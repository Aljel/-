```Go
package internal

import "github.com/gin-gonic/gin"

func Launch() {
	gin.SetMode("debug")
	
	r := gin.New()
	
	srv := &http.Server
	
	listener, err := net.Listen("tcp", fmt.Sprintf(
		"%s:%d",
		"localhost",
		9000,
	))
	if err != nil {
		fmt.Println("could not create listener", err)
		return
	}
	defer listener.Close
	
	r := gin.New()
	r.GET("/ping", func(c *gin.Context) {
		c.String(gttp.StatusOK, "pong")
	})
	
	if err := srv.Serve(listener); err != nil && err != http.ErrServerClosed {
		fmt.Println("Could not serve", err)
	}
	
	_ = r.Run()
}
```
