```js
function createUser(name, age) {
	if (name.trim() === "") {
		throw {
			name: "NameIsEmpty",
			message: "Имя не может быть пустым"
		}
	}
	return {name, age}
}

try {
	const { name } = createUser(
		prompt("Введите имя"),
		parseInt(prompt("Введите возраст"))
	);
	console.log(`Привет, ${name}!`)
} catch (e) {
	console.error(e.message);
}
```
